Getting rid of your for `for-loops` in javascript.

{% highlight js %} var original = [1,2,3,4,5,6]; var i, newList = []; for (i = 0; i < original.length; i++) { newList.push(original[i] * 2); } {% endhighlight %}

{% highlight js %} var original = [1,2,3,4,5,6]; var newList = original.map(function(o){ return o * 2; }); {% endhighlight %}
