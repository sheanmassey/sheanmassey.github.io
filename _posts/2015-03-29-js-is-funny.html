Keep it hidden. You shouldn't pollute the global scope unless necessary.

{% highlight js %} (function(){ var your_script; })(); {% endhighlight %}

Using the same 'immediately invoked function' you can get a hold of an undefined variable which is `undefined`

{% highlight js %} (function(undefined){ var your_script; // now you compare for `undefined` using: if (your_script === undefined){ // ... } })(); {% endhighlight %}

Map and Reduce:

{% highlight js %} var l = [21,49,80,43,59]; var doubled = l.map(function(d){ return d*2; }); var max_value = doubled.reduce(function(a,b){ return a > b ? a : b; }, 0); {% endhighlight %}

Augmenting an array of elements with forEach

{% highlight js %} var l = [ {id: 3, first_name: 'john', last_name: 'doe'}, {id: 4, first_name: 'bill', last_name: 'masset'} ]; l.forEach(function(o){ o.full_name = o.first_name + ' ' + o.last_name; }); {% endhighlight %}

Objects, the right way

{% highlight js %} var object_factory = function(params){ var hidden_property = 42; var built_with_params, etc; var hidden_func = function(m,n){ return m > n; }; return { public_method: function(){ return hidden_property; }, another: function(m,n){ hidden_func(m,n); } } }; {% endhighlight %}

Arrays are objects, too

{% highlight js %} var l = [1,2,3,4,5]; l.total = l.reduce(function(a,b){ return a + b; }, 0); {% endhighlight %}

So comparing equality also acts like objects, not primitive lists
